def process_lightcurve_single_band(per, idx, fid=1):
    light_curve, period, label = ztf_subset.get_lc(idx)
    mjds, mags, errs, fids = light_curve
    mjd, mag, err = mjds[fids==fid], mags[fids==fid], errs[fids==fid]
    per.set_data(mjd, mag, err)        
    per.frequency_grid_evaluation(fmin=0.1, fmax=10, fresolution=0.001)
    freq, pers = per.get_periodogram()
    return freq, pers, mjd, mag, err
